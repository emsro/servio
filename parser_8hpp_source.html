59struct lexer
60{
61 char const* p;
62 char const* e;
63
64 lexer( std::string_view sv )
65 : p( sv.data() )
66 , e( sv.data() + sv.size() )
67 {
68 }
69
70 vari::vval< tok, end > next()
71 {
72 while ( p != e && str_lib::bits::is_ws( *p ) )
73 p++;
74
75 if ( p == e || *p == '\0' )
76 return end{};
77 str_lib::s_to_nr_res r;
78 if ( str_lib::s_to_nr( p, e, r ) ) {
79 if ( r.is_num )
80 return r.n;
81 return r.r;

83 if ( auto s = parse_str( p, e ) )
84 return str{ .s = *s };
85 char const* pp = p;
86 if ( str_lib::lex_letters( p, e ) )
87 return id{ .s = { pp, p } };
88 return err{};
89 }
