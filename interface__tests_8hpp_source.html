24// Test that clock of chip works correctly:
25// - time increases
26struct clock_test
27{
28 clk_iface& clk;
29
30 std::string_view name = "clock_test";
31
32 t::coroutine< void > run( auto&, ftest::uctx& ctx )
33 {
34 std::size_t wait_cycles = 100;
35 microseconds t1 = clk.get_us();
36 for ( std::size_t i = 0; i < wait_cycles; i++ )
37 asm( "nop" );
38 microseconds t2 = clk.get_us();
39 co_await ctx.expect( t2 > t1 );
40 ctx.coll.set( 0, "t1", t1 );
41 ctx.coll.set( 0, "t2", t2 );
42 ctx.coll.set( 0, "wait cycles", wait_cycles );
43
44 microseconds now = clk.get_us();
45 microseconds last = now;
46 microseconds end = now + 2_s;
47 do {
48 last = now;
49 now = clk.get_us();
50 if ( last > now ) {
51 ctx.coll.set( "last", last );
52 ctx.coll.set( "now", now );
53 }
54 co_await ctx.expect( last <= now );
55 } while ( clk.get_us() < end );
56 }
57};
