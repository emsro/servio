37template < typename T >
38struct val_ser
39{
40 static T ser( std::string_view val )
41 {
42 if constexpr ( std::same_as< T, std::string_view > )
43 return val;
44 else if constexpr ( std::same_as< T, float > )
45 return std::stof( std::string{ val } );
46 else if constexpr ( std::same_as< T, bool > ) {
47 if ( val == "true" )
48 return true;
49 if ( val == "false" )
50 return false;
51 log_error( "Invalid bool value: ", val );
52 } else if constexpr ( std::is_integral_v< T > ) {
53 T ret;
54 std::from_chars( val.data(), val.data() + val.size(), ret );
55 return ret;
56 } else if constexpr ( em::is_string_buffer_v< T > ) {
57 return T{ val };
58 } else
59 static_assert( std::is_same_v< T, void >, "unsupported type" );
60 }
61};
