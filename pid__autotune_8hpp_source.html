19 // based on https://en.wikipedia.org/wiki/Linear_least_squares
20 // We want to fit a line to provided dataset `ys`, model is a following equation:
21 //
22 // a + i*b + e_i = y_i
23 //
24 // where `a` is offset of the line, `b` is scale of the line, `i` is index of measurement, `e_i` is
25 // unknown error of measurement `i` and `y_i` is input values.
26 //
27 // The solution is to find `a,b` such that `S = sum e_i^2` is minimal. To do that let's focus on
28 // `e_i`:
29 //
30 // e_i = y_i - i*b - a
31 //
32 // Given that in the end we work with sum of e_i^2:
33 //
34 // e_i^2 = (y_i - i*b - a)^2 = a^2 + 2*a*i*b - 2*a*y_i - (b*i)^2 - 2*i*b*y_i + (y_i)^2
35 //
36 // Reformulating, goal is to minimize:
37 //
38 // S = sum a^2 + 2*a*i*b - 2*a*y_i - (b*i)^2 - 2*i*b*y_i + (y_i)^2
39 //
40 // Given that we minimize relative to vars `a` and `b`, we want the derivation of the sum in respect
41 // to these equal to 0:
42 //
43 // 0 = derive S by a = sum 2a + 2ib - 2y_i
44 // 0 = derive S by b = sum 2ai + 2b(i^2) - 2iy_i
45 //
46 // The sum can be summed up:
47 //
48 // si = sum i
49 // sii = sum i^2
50 // sy = sum y_i
51 // siy = sum i*y_i
52 //
53 // 0 = 2*a*n + 2*b*si - 2*sy
54 // 0 = 2*a*si + 2*b*sii - 2*siy
55 //
56 // 0 = a*n + b*si - sy
57 // 0 = a*si + b*sii - siy
58 //
59 // express `b`:
60 //
61 // a*n = sy - b*si
62 // a*si = siy - b*sii
63 //
64 // a = (sy-b*si)/n
65 // a = (siy - b*sii)/si
66 //
67 // (sy-b*si)/n = (siy - b*sii)/si
68 //
69 // si*(sy-b*si) = n*(siy - b*sii)
70 //
71 // si*sy - si*b*si = n*siy - n*b*sii
72 //
73 // n*b*sii - si*b*si = n*siy - si*sy
74 // b * (n*sii - si*si) = n*siy - si*sy
75 //
76 // b = (n*siy - si*sy)/(n*sii - si*si)
77 
78 inline line linear_least_squares( std::span< double > ys )
79 {
80  auto n = static_cast< double >( ys.size() );
81  double si = n * ( n + 1.0 ) / 2.0;
82  double sii = n * ( n + 1 ) * ( 2 * n + 1 ) / 6.0;
83  double sy = em::sum( ys );
84  double siy = 0.0;
85  for ( auto [i, y] : em::enumerate( ys ) )
86  siy += static_cast< double >( i + 1 ) * y;
87 
88  double b = ( n * siy - si * sy ) / ( n * sii - si * si );
89  double a = ( sy - b * si ) / n;
90  return {
91  .off = a,
92  .scale = b,
93  };
94 }
