20// based on https://en.wikipedia.org/wiki/Linear_least_squares
21// We want to fit a line to provided dataset `ys`, model is a following equation:
22//
23// a + i*b + e_i = y_i
24//
25// where `a` is offset of the line, `b` is scale of the line, `i` is index of measurement, `e_i` is
26// unknown error of measurement `i` and `y_i` is input values.
27//
28// The solution is to find `a,b` such that `S = sum e_i^2` is minimal. To do that let's focus on
29// `e_i`:
30//
31// e_i = y_i - i*b - a
32//
33// Given that in the end we work with sum of e_i^2:
34//
35// e_i^2 = (y_i - i*b - a)^2 = a^2 + 2*a*i*b - 2*a*y_i - (b*i)^2 - 2*i*b*y_i + (y_i)^2
36//
37// Reformulating, goal is to minimize:
38//
39// S = sum a^2 + 2*a*i*b - 2*a*y_i - (b*i)^2 - 2*i*b*y_i + (y_i)^2
40//
41// Given that we minimize relative to vars `a` and `b`, we want the derivation of the sum in respect
42// to these equal to 0:
43//
44// 0 = derive S by a = sum 2a + 2ib - 2y_i
45// 0 = derive S by b = sum 2ai + 2b(i^2) - 2iy_i
46//
47// The sum can be summed up:
48//
49// si = sum i
50// sii = sum i^2
51// sy = sum y_i
52// siy = sum i*y_i
53//
54// 0 = 2*a*n + 2*b*si - 2*sy
55// 0 = 2*a*si + 2*b*sii - 2*siy
56//
57// 0 = a*n + b*si - sy
58// 0 = a*si + b*sii - siy
59//
60// express `b`:
61//
62// a*n = sy - b*si
63// a*si = siy - b*sii
64//
65// a = (sy-b*si)/n
66// a = (siy - b*sii)/si
67//
68// (sy-b*si)/n = (siy - b*sii)/si
69//
70// si*(sy-b*si) = n*(siy - b*sii)
71//
72// si*sy - si*b*si = n*siy - n*b*sii
73//
74// n*b*sii - si*b*si = n*siy - si*sy
75// b * (n*sii - si*si) = n*siy - si*sy
76//
77// b = (n*siy - si*sy)/(n*sii - si*si)
78
79inline line linear_least_squares( std::span< double > ys )
80{
81 auto n = static_cast< double >( ys.size() );
82 double si = n * ( n + 1.0 ) / 2.0;
83 double sii = n * ( n + 1 ) * ( 2 * n + 1 ) / 6.0;
84 double sy = em::sum( ys );
85 double siy = 0.0;
86 for ( auto [i, y] : em::enumerate( ys ) )
87 siy += static_cast< double >( i + 1 ) * y;
88
89 double b = ( n * siy - si * sy ) / ( n * sii - si * si );
90 double a = ( sy - b * si ) / n;
91 return {
92 .off = a,
93 .scale = b,
94 };
95}

126 std::span< double > lls_span{

128 std::cout << "lls_span size: " << lls_span.size() << std::endl;
129 line guide = linear_least_squares( lls_span );

131 double K = range;
132 double L = -guide.off;
133 // y = a + b*x
134 // x = (y-a)/b
135 double T = ( K - guide.off ) / guide.scale - L;
136
137 std::cout << "K: " << K << "\tL: " << L << "\tT: " << T << std::endl;
138
139 // Cohen coon, inspiration:
140 // https://www.incatools.com/pid-tuning/pid-tuning-methods/
141 double tau = L / ( L + T );
142 double a = K * L / T;
143
144 double K_p = ( 1.35 / a ) * ( 1 + 0.18 * tau / ( 1 - tau ) );
145 double T_i = L * ( 2.5 - 2.0 * tau ) / ( 1 - 0.39 * tau );
146 double T_d = L * ( 0.37 - 0.37 * tau ) / ( 1 - 0.81 * tau );
147 double K_i = K_p / T_i;
148 double K_d = K_p * T_d;
149 std::cout << "P: " << K_p << "\tI: " << K_i << "\tD: " << K_d << std::endl;
150}
