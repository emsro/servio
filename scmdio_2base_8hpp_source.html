169template <>
170struct std::formatter< nlohmann::json, char >
171{
172 bool pretty = false;
173
174 template < class ParseContext >
175 constexpr ParseContext::iterator parse( ParseContext& ctx )
176 {
177 auto it = ctx.begin();
178 if ( it == ctx.end() )
179 return it;
180
181 if ( *it == '#' ) // Use `#` to indicate pretty formatting
182 {
183 pretty = true;
184 ++it;
185 }
186 if ( it != ctx.end() && *it != '}' )
187 throw std::format_error( "Invalid format args for nlohmann::json." );
188
189 return it;
190 }
191
192 template < class FmtContext >
193 FmtContext::iterator format( nlohmann::json const& j, FmtContext& ctx ) const
194 {
195 std::ostringstream out;
196 if ( pretty )
197 out << std::setw( 4 ) << j;
198 else
199 out << j; // Compact JSON
200
201 return std::ranges::copy( out.str(), ctx.out() ).out;
202 }
