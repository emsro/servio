15inline vari::vval< vari::vref< iface >, str_err >
16find_iface( iface* root_handler, gov::governor_manager& gm, std::string_view gov )
17{
18 if ( gov == "" ) {
19 if ( root_handler )
20 return vari::vref< iface >{ *root_handler };
21 return "unknown governor"_err;
22 }
23 for ( vari::vref< gov::governor > g : gm.governors() )
24 if ( g->name() == gov ) {
25 auto* c = g->get_cfg();
26 if ( c )
27 return vari::vref< iface >{ *c };
28 return "governor without config"_err;
29 }
30 return "unknown governor"_err;
31}

39inline opt< resolve_governor_res >
40resolve_governor( std::string_view field, std::string_view governor )
41{
42 auto npos = field.find( '.' );
43 if ( npos != std::string_view::npos ) {
44 if ( !governor.empty() )
45 return {};
46 return resolve_governor_res{
47 .governor = field.substr( 0, npos ), .field = field.substr( npos + 1 ) };
48 } else {
49 return resolve_governor_res{ .governor = governor, .field = field };
50 }
51}

80struct dispatcher
81{
82 cfg::iface* root_handler;
83 gov::governor_manager& gov;

85 // Dispatcher honors advanced syntax, there are two ways to set field of governor:
86 // 1) Using dot syntax: "cfg set governor_name.field_name value"
87 // 2) Using governor argument: "cfg set field_name value governor=governor_name"
88 // @param field is either name of param or dot syntax with governor name
89 // @param governor is name of explicit governor parameter or empty string
90
91 opt_str_err on_cmd_set(
92 std::string_view field,
93 vari::vref< servio::cfg::base_t const > value,
94 std::string_view governor )
95 {
96 auto r = resolve_governor( field, governor );
97 if ( !r )
98 return "dot syntax collides with governor"_err;
99
100 return find_iface( root_handler, gov, r->governor )
101 .visit(
102 [&]( vari::vref< iface > ifc ) -> opt_str_err {
103 return ifc->on_cmd_set( r->field, value );
104 },
105 [&]( str_err const& e ) -> opt_str_err {
106 return e;
107 } );
108 }
109
110 vari::vval< base_t, str_err >
111 on_cmd_get( std::string_view field, std::string_view governor )
112 {
113 auto r = resolve_governor( field, governor );
114 if ( !r )
115 return "dot syntax collides with governor"_err;
116
117 using R = vari::vval< base_t, str_err >;
118 return find_iface( root_handler, gov, r->governor )
119 .visit(
120 [&]( vari::vref< iface > ifc ) -> R {
121 return ifc->on_cmd_get( r->field );
122 },
123 [&]( str_err const& e ) -> R {
124 return e;
125 } );
126 }
127
128 vari::vval< std::string_view, str_err >
129 on_cmd_list( int32_t offset, std::string_view governor )
130 {
131 using R = vari::vval< std::string_view, str_err >;
132 return find_iface( root_handler, gov, governor )
133 .visit(
134 [&]( vari::vref< iface > ifc ) -> R {
135 return ifc->on_cmd_list( offset );
136 },
137 [&]( str_err const& e ) -> R {
138 return e;
139 } );
140 }
