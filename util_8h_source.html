42inline vari::vopt< num > hex_numeric( char const*& p, char const* e )
43{
44 SERVIO_NUM_ASSERT( p != e );
45 num x = 0;
46 if ( !_is_hex( *p ) )
47 return {};
48 auto f = [&]( char c ) {
49 x *= 16;
50 if ( c > 'a' )
51 x += 10 + ( c - 'a' );
52 else if ( c > 'A' )
53 x += 10 + ( c - 'A' );
54 else
55 x += c - '0';
56 };
57 do
58 f( *p++ );
59 while ( p != e && _is_hex( *p ) );
60 return x;
61}

89inline vari::vopt< num, real > numreal( char const*& p, char const* e )
90{
91 SERVIO_NUM_ASSERT( p != e );
92 if ( hex_prefix( p, e ) ) {
93 p++;
94 return hex_numeric( ++p, e );
95 }
96 int sign = 1;
97 if ( *p == '-' ) {
98 p++;
99 sign = -1;
100 }
101 if ( p == e )
102 return {};
103 num x = 0;
104 if ( *p != '.' ) {
105 if ( auto r = dec_numeric( p, e ) )
106 x = *r;
107 else
108 return {};
109 }
110 if ( *p != '.' )
111 return sign * x;
112 float y = 0.0F;
113 float exp = 1.0F;
114 while ( ++p != e && _is_dec( *p ) ) {
115 exp *= 10.0F;
116 y *= 10;
117 y += *p - '0';
118 }
119 return (float) sign * ( (float) x ) + ( y / exp );
120}
